   146→   146→-- User profiles (auto-created on signup via trigger below)

   156→   156→  id               UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
   157→   157→  user_id          UUID REFERENCES public.profiles(id) ON DELETE SET NULL,
   158→   158→  target_role      TEXT NOT NULL,
   159→   159→  readiness_score  INTEGER CHECK (readiness_score BETWEEN 0 AND 100),
   160→   160→  score_label      TEXT,
   161→   161→  gaps             JSONB DEFAULT '[]',
   162→   162→  strengths        JSONB DEFAULT '[]',
   163→   163→  roadmap          JSONB DEFAULT '[]',
   164→   164→  created_at       TIMESTAMPTZ DEFAULT NOW()
   165→   165→);
   166→   166→
   167→   167→-- Anonymized skill trend tracking (no PII)
   168→   168→CREATE TABLE public.skill_trends (
   169→   169→  id           UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
   170→   170→  target_role  TEXT NOT NULL,
   171→   171→  skill_name   TEXT NOT NULL,
   172→   172→  frequency    INTEGER DEFAULT 1,
   173→   173→  last_seen    TIMESTAMPTZ DEFAULT NOW(),
   174→   174→  UNIQUE (target_role, skill_name)
   175→   175→);
   176→   176→
   177→   177→-- Row-level security
   178→   178→ALTER TABLE public.profiles     ENABLE ROW LEVEL SECURITY;
   179→   179→ALTER TABLE public.analyses     ENABLE ROW LEVEL SECURITY;
   180→   180→ALTER TABLE public.skill_trends ENABLE ROW LEVEL SECURITY;
   181→   181→
   182→   182→-- Policies
   183→   183→CREATE POLICY "Own profile readable"
   184→   184→  ON public.profiles FOR SELECT USING (auth.uid() = id);
   185→   185→
   186→   186→CREATE POLICY "Own analyses readable"
   187→   187→  ON public.analyses FOR SELECT USING (auth.uid() = user_id);
   188→   188→
   189→   189→CREATE POLICY "Own analyses insertable"
   190→   190→  ON public.analyses FOR INSERT WITH CHECK (auth.uid() = user_id);
   191→   191→
   192→   192→CREATE POLICY "Skill trends public read"
   193→   193→  ON public.skill_trends FOR SELECT USING (true);
   194→   194→
   195→   195→-- Auto-create profile on Supabase Auth signup
   196→   196→CREATE OR REPLACE FUNCTION public.handle_new_user()
   197→   197→RETURNS TRIGGER AS $$
   198→   198→BEGIN
   199→   199→  INSERT INTO public.profiles (id, email, full_name)
   200→   200→  VALUES (
   201→   201→    NEW.id,
   202→   202→    NEW.email,
   203→   203→    COALESCE(NEW.raw_user_meta_data->>'full_name', 'Anonymous')
   204→   204→  );
   205→   205→  RETURN NEW;
   206→   206→END;
   207→   207→$$ LANGUAGE plpgsql SECURITY DEFINER;

   209→   209→CREATE OR REPLACE TRIGGER on_auth_user_created
   210→   210→  AFTER INSERT ON auth.users
   211→   211→  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();
   212→   212→```
